corr_matrix <- cor(numeric_data, use = "complete.obs")

get_lower_tri <- function(corr_matrix) {
  corr_matrix[upper.tri(corr_matrix)] <- NA
  return(corr_matrix)
}

corr_mat <- melt(get_lower_tri(corr_matrix), na.rm = TRUE)
